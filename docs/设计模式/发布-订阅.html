异步通信

function Event() {
  this.obj = {};
}

Event.prototype.on = function (name, cb) {
  if (!this.obj[name]) {
    this.obj[name] = [];
  }
  this.obj[name].push(cb);
};

Event.prototype.emit = function (name, ...args) {
  let arr = this.obj[name];
  arr?.forEach((item) => item(...args));
};

let e = new Event();
e.on("ok", function (arg1, arg2) {
  console.log("成功触发了", arg1, arg2);
});
e.emit("ok", "简单的参数啊");
